function repeat(func, times, wait) {
    const delay = time => new Promise(resolve => setTimeout(resolve, time));
    return function(str) {
        let i = 0;
        while(i < times) {
            delay(wait * i).then(() => {
                func(str);
            })
            i++;
        }
    }
}

const repeatFun = repeat(console.log, 6,1000)
repeatFun("hello world");
